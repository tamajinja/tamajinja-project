    file_stem = file_path.stem

    # 関連する profiling 情報を探す

    for key, text in profiling_texts.items():
        if file_stem in key:
            merged_text += f"\n\n---\n\n### 🔍 付録: {key}\n\n{text}"
        elif any(file_stem in key.replace(suffix, "") for suffix in suffixes):
            merged_text += f"\n\n---\n\n### 🔍 付録: {key}\n\n{text}"

    # 書き出し
    output_path = Path(output_dir) / file_path.name
    with open(output_path, 'w') as f:
        f.write(merged_text)
